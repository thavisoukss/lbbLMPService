# Table: LMPS_TXN_DETAIL

Interim replacement for `WITHDRAW_TXN`. Records every outward transfer transaction (QR, Account, Bio) initiated by a customer. A row is inserted at inquiry time (`STATUS = DEBIT_PENDING`) alongside the corresponding `WITHDRAW_TXN` row and updated to `COMPLETED` when the CBS transfer succeeds. This table will eventually become the sole record of transfer transactions once the dual-write migration is complete and `WITHDRAW_TXN` is retired.

## STATUS Values

| Value | Meaning |
|-------|---------|
| `DEBIT_PENDING` | Initial state — debit not yet posted to CBS |
| `DEBITED` | Debit posted successfully to CBS |
| `CREDIT_PENDING` | Waiting for provider to confirm credit |
| `CREDITED` | Credit confirmed by provider |
| `COMPLETED` | Both debit and credit completed |
| `REVERSAL_PENDING` | Reversal requested, pending execution |
| `REVERSAL` | Reversal completed |
| `REVERSAL_FAILED` | Reversal attempted but failed |
| `FAILED` | Transaction failed (see `ERROR_CODE` / `ERROR_MESSAGE`) |
| `CANCELLED` | Transaction cancelled before processing |

## DDL

-- Sequence
CREATE SEQUENCE LMPS_TXN_DETAIL_SEQ START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

-- Table
CREATE TABLE LMPS_TXN_DETAIL (
    ID                         NUMBER          NOT NULL,
    PAYMENT_CHANNEL_ID         NUMBER          NOT NULL,
    CUSTOMER_ID                VARCHAR2(50)    NOT NULL,
    TRANSACTION_ID             VARCHAR2(100)   NOT NULL,

    DR_ACCOUNT_NO              VARCHAR2(50)    NOT NULL,
    DR_CIF                     VARCHAR2(20),
    CR_CIF                     VARCHAR2(20),
    CR_ACCOUNT_NO              VARCHAR2(50),
    AMOUNT                     NUMBER(18,4)    NOT NULL,
    FEE_AMT                    NUMBER(18,4)    DEFAULT 0 NOT NULL,
    FEE_PROVIDER_AMT           NUMBER(18,4)    DEFAULT 0 NOT NULL,
    CURRENCY_CODE              VARCHAR2(3)     DEFAULT 'LAK' NOT NULL,
    FEE_CURRENCY_CODE          VARCHAR2(3)     DEFAULT 'LAK' NOT NULL,
    FEE_PROVIDER_CURRENCY_CODE VARCHAR2(3)     DEFAULT 'LAK' NOT NULL,
    TRAN_TYPE                  VARCHAR2(20),
    CR_GL_CODE                 VARCHAR2(20),
    DR_SETTLE_MTD              VARCHAR2(20),
    CR_SETTLE_MTD              VARCHAR2(20),
    RUN_DATE                   TIMESTAMP(6) WITH TIME ZONE,
    CORE_BANKING_REF           VARCHAR2(100),
    TRANSACTION_DATE           TIMESTAMP(6) WITH TIME ZONE,
    ERROR_CODE                 VARCHAR2(50),
    ERROR_MESSAGE              VARCHAR2(500),
    CREATED_AT                 TIMESTAMP(6)    DEFAULT SYSTIMESTAMP NOT NULL,
    UPDATED_AT                 TIMESTAMP(6),
    DR_SEQ_NO                  NUMBER,
    CR_SEQ_NO                  NUMBER,
    PUBLIC_KEY                 VARCHAR2(200),
    DR_ACCOUNT_NAME            VARCHAR2(200),
    CR_ACCOUNT_NAME            VARCHAR2(200),
    VERSION                    NUMBER          DEFAULT 1 NOT NULL,
    DELETE_AT                  TIMESTAMP(6),
    REMARK                     VARCHAR2(100),
    FEE_LIST                   VARCHAR2(4000),
    CLIENT_REF                 VARCHAR2(100),
    CONSTRAINT PK_LMPS_TXN_DETAIL          PRIMARY KEY (ID),
    CONSTRAINT UK_LMPS_TXN_DETAIL_ID       UNIQUE (TRANSACTION_ID),
    CONSTRAINT UK_LMPS_TXN_DETAIL_NONCE    UNIQUE (NONCE),
    CONSTRAINT UK_LMPS_TXN_DETAIL_PROV_REF UNIQUE (PROV_REF_NO),
    CONSTRAINT UK_LMPS_TXN_DETAIL_CORE_REF UNIQUE (CORE_BANKING_REF),
    CONSTRAINT CHK_LMPS_TXN_DETAIL_STATUS  CHECK (STATUS IN (
        'DEBIT_PENDING','DEBITED','CREDIT_PENDING','CREDITED','COMPLETED',
        'REVERSAL_PENDING','REVERSAL','REVERSAL_FAILED','FAILED','CANCELLED'

CREATE INDEX IDX_LMPS_TXN_DETAIL_DR_ACCT       ON LMPS_TXN_DETAIL (DR_ACCOUNT_NO);
CREATE INDEX IDX_LMPS_TXN_DETAIL_CR_CIF        ON LMPS_TXN_DETAIL (CR_CIF);

CREATE INDEX IDX_LMPS_TXN_DETAIL_STATUS        ON LMPS_TXN_DETAIL (STATUS, CREATED_AT);

CREATE INDEX IDX_LMPS_TXN_DETAIL_CREATED_DESC  ON LMPS_TXN_DETAIL (CREATED_AT DESC);

-- Triggers
CREATE OR REPLACE TRIGGER TR_LMPS_TXN_DETAIL_BI
BEFORE INSERT ON LMPS_TXN_DETAIL
FOR EACH ROW
BEGIN
    IF :NEW.ID IS NULL THEN
        SELECT LMPS_TXN_DETAIL_SEQ.NEXTVAL INTO :NEW.ID FROM DUAL;
    END IF;
END;
/

CREATE OR REPLACE TRIGGER TR_LMPS_TXN_DETAIL_BU
BEFORE UPDATE ON LMPS_TXN_DETAIL
FOR EACH ROW
BEGIN
    :NEW.UPDATED_AT := SYSTIMESTAMP;
END;
/

CREATE OR REPLACE TRIGGER TR_LMPS_TXN_DETAIL_VERSION
BEFORE UPDATE ON LMPS_TXN_DETAIL
FOR EACH ROW
BEGIN
    :NEW.VERSION := :OLD.VERSION + 1;
END;
/
